// Modal Functionality function showModal(modalId) { document.getElementById(modalId).style.display = 'flex'; } function closeModal(modalId) { document.getElementById(modalId).style.display = 'none'; } $(document).ready(function () { // Show Modals $('#product-btn').click(function () { showModal('product-modal'); }); $('#add-cashier-btn').click(function () { showModal('add-cashier-modal'); }); $('#calculator-btn').click(function () { showModal('calculator-modal'); }); // Close Modals $('#close-product-modal').click(function () { closeModal('product-modal'); }); $('#close-add-cashier-modal').click(function () { closeModal('add-cashier-modal'); }); $('#close-calculator-modal').click(function () { closeModal('calculator-modal'); }); }); // AJAX for Adding Product $('#add-product-form').on('submit', function (e) { e.preventDefault(); let formData = new FormData(this); $.ajax({ url: '/api/products', type: 'POST', data: formData, processData: false, contentType: false, success: function (response) { alert('Product added successfully!'); closeModal('product-modal'); }, error: function () { alert('Error adding product. Please try again.'); } }); }); // AJAX for Adding Cashier $('#add-cashier-form').on('submit', function (e) { e.preventDefault(); $.ajax({ url: '/api/cashiers', type: 'POST', data: $(this).serialize(), success: function (response) { alert('Cashier added successfully!'); closeModal('add-cashier-modal'); }, error: function () { alert('Error adding cashier. Please try again.'); } }); }); // Calculator Logic $('#calculator-form').on('submit', function (e) { e.preventDefault(); let productName = $('#calculator-product-name').val(); let quantity = $('#calculator-quantity').val(); let price = $('#calculator-price').val(); let discount = $('#calculator-discount').val(); // Display Results $('#display-product-name').text(productName); $('#display-quantity').text(quantity); $('#display-price').text(price); $('#display-discount').text(discount); let total = (price * quantity) - ((price * quantity) * (discount / 100)); $('#result').text('Total Cost: $' + total.toFixed(2)); }); // Additional AJAX Features // AJAX for Orders, Inventory, and Users (Not fully implemented, just structure) function loadTableData(endpoint, tableId) { $.ajax({ url: endpoint, type: 'GET', success: function (data) { let tableBody = $(tableId + ' tbody'); tableBody.empty(); data.forEach(function (item) { let row = ''; Object.values(item).forEach(function (value) { row += '' + value + ''; }); row += ''; tableBody.append(row); }); }, error: function () { alert('Error loading data. Please try again.'); } }); } // Example usage: // loadTableData('/api/orders', '#orders-table'); // loadTableData('/api/inventory', '#inventory-table'); // loadTableData('/api/users', '#users-table');
